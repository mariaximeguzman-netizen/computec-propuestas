async function generarPropuesta(){

const apiKey = document.getElementById('api-key-input').value.trim();

if(!apiKey){

alert('Por favor ingresa la clave de acceso Computec para continuar.');

return;

}

document.getElementById('gen-inicial').style.display='none';

document.getElementById('gen-loading').style.display='block';

document.getElementById('gen-resultado').style.display='none';

document.getElementById('gen-error').style.display='none';

try{

const prompt=buildPromptText();

const response=await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`,{

method:'POST',

headers:{'Content-Type':'application/json'},

body:JSON.stringify({

contents:[{parts:[{text:prompt}]}],

generationConfig:{temperature:0.7,maxOutputTokens:8192}

})

});

if(!response.ok) throw new Error('Error API: '+response.status);

const data=await response.json();

const texto=data.candidates?.[0]?.content?.parts?.[0]?.text||'';

if(!texto) throw new Error('Sin respuesta');

propuestaGenerada=texto;

document.getElementById('resultado-propuesta').textContent=texto;

document.getElementById('gen-loading').style.display='none';

document.getElementById('gen-resultado').style.display='block';

} catch(e){

document.getElementById('gen-loading').style.display='none';

document.getElementById('gen-error').style.display='block';

console.error(e);

}

}
